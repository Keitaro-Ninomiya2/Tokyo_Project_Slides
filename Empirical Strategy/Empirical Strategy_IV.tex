\begin{frame}{Identification Strategy}
Estimated at the employee level.
\begin{gather*}
\underbrace{Y_{i,t}}_{Present Diversity} = \underbrace{X_{i,j,p}}_{Past Diversity} \beta + \iota_j + \mu_p + \epsilon_{i,j,p,t} \\
    X_{i,j,p} = Z_{i,j,p} \delta + \iota_j + \mu_p + \varsigma_{i,j,p,t}
\end{gather*}
\begin{itemize}
    \item \( Y_{i,t} \): Share of female employees at time \(t\) for employee $i$.
    \item \( X_{i,j,p} \): Mean share of female co-workers during war period.
    \item \( Z_{i,j,p} \): Share of Coworkers drafted during war period.
    \item \( \iota_j, \mu_p\): During-war division and occupation fixed effects.
\end{itemize}
\end{frame}
