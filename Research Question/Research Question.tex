\begin{frame}{Research Question}
    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Research Question]
        \begin{itemize}
                    \item Did employees w. co-working experience with women during the war continue to work w. women?
        \end{itemize}
    \end{tcolorbox}
    \pause
    \begin{tcolorbox}[colback=green!5!white,colframe=green!75!black,title=Method]
        \begin{itemize}
            \item Data: Newly constructed data on the universe of public sector employees in 1928-1963 with positional/occupational details (n=330,000).

            \pause
            \item Method: Regress gender-unbalance across offices in post-war periods by constituents war-time experience with working with women.

            \pause
            \item Identification: Use office level gender imbalance during war periods induced by military drafting.

            $\Rightarrow$ IV strategy.
        \end{itemize}
    \end{tcolorbox}
\end{frame}
